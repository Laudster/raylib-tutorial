#include "raylib.h"
#include <stdlib.h.hh>
#include <time.h.hh>

int main()
{
    InitWindow(1200, 800, "game");
    SetWindowState(FLAG_WINDOW_UNDECORATED);
    SetTargetFPS(60);

    srand(time(NULL));

    int playerPos = 400;
    int score = 0;

    int enemyX = 1300;
    int enemyY = rand() % (700 - 100 + 1);

    while (!WindowShouldClose())
    {

        if (IsKeyDown(KEY_SPACE)) playerPos -= 10;

        playerPos += 5;
        enemyX -= 5;

        if (enemyX < -50)
        {
            enemyY = rand() % (400 - 50 + 1);
            enemyX = 1300;
        }

        if (enemyX == 400 && playerPos < enemyY | enemyX == 400 && playerPos > enemyY + 200)
        {
            score = 0;
            enemyY = rand() % (400 - 50 + 1);
            enemyX = 1300;
        }

        if (enemyX == 200) score++;

        BeginDrawing();
            ClearBackground(RAYWHITE);
            DrawText(TextFormat("%i", score), 500, 200, 300, YELLOW);

            DrawCircle(200, playerPos, 30, BLUE);

            DrawRectangle(enemyX, 0, 50, enemyY, BLACK);
            DrawRectangle(enemyX, enemyY + 200, 50, 800, BLACK);
        EndDrawing();
    }

    CloseWindow();

    return 0;
}
